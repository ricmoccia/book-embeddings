268function getGraphFromFileString(fileString) {
269 const G = { nodes: [], edges: [] };
270 let i = 0;
271 let current = "";
272 let readingNodes = true;
273 let sourceNode = -1;
274
275 while (i < fileString.length) {
276 if (fileString[i] == '\n') {
277 if (readingNodes) {
278 G.nodes.push({id: current, label: current});
279 } else {
280 G.edges.push({from: sourceNode, to: current});
281 }
282 current = "";
283 } else if (fileString[i] == ',') {
284 sourceNode = current;
285 current = "";
286 readingNodes = false;
287 } else if (fileString[i] != '\r') {
288 current = current + fileString[i];
289 }
290 i++;
291 }
292 if (current.length > 0) {
293 if (readingNodes) {
294 G.nodes.push({id: current, label: current});
295 } else {
296 G.edges.push({from: sourceNode, to: current});
297 }
298 }
299 return G;
300}

302// =============================================================================
303// Parsing of messages emitted by the WebAssembly module
304// =============================================================================
305//
306// The C++ side writes various kinds of lines to stdout. They are forwarded
307// here by the patch in update_binary.py (which replaces the `out` callback
308// with `getDataFromWasm` inside put_char). The recognized prefixes are:
309//
310// "NUMBER OF LAYOUTS: <n>" -> updates numberOfLayouts
311// "RESULT: <v0> <v1> ..." -> pushes a new layout
312//
313// "BLOCKS_BEGIN count=<n>" -> begin block decomposition
314// "BLOCK_EDGE block=<i> from=<u> to=<v>"
315// "BLOCKS_END" -> end block decomposition
316// (emitted ONCE per WASM session,
317// blocks are a static property of
318// the input graph)
319//
320// "BCT_BEGIN root=<id>" -> begin BCT description
321// "BCT_NODE id=<id> kind=<block|cutpoint> value=<v>"
322// "BCT_EDGE from=<id> to=<id> type=<0|1|2>"
323// "BCT_END" -> end BCT description
324//
325// "FPQ_BEGIN root=<id>" -> begin FPQ-tree description
326// "FPQ_NODE id=<id> type=<P|F_BLOCK|F_GADGET|LEAF> [block=<i>] [value=<v>]"
327// "FPQ_EDGE from=<id> to=<id> pos=<k>" -> pos = child index inside its parent
328// "FPQ_END" -> end FPQ-tree description
329//
330// Everything else is debug output and ends up only in console.log.
331// =============================================================================

338function getDataFromWasm(message) {
339 console.log(message);
340
341 if (typeof message !== 'string') return;
342
343 // --- Blocks (biconnected decomposition, emitted once) ---
344 if (message.startsWith('BLOCKS_BEGIN')) {
345 const m = message.match(/count=(\d+)/);
346 const count = m ? parseInt(m[1], 10) : 0;
347 currentBlocks = [];
348 for (let i = 0; i < count; i++) currentBlocks.push([]);
349 return;
350 }
351 if (message.startsWith('BLOCK_EDGE')) {
352 if (!currentBlocks) return;
353 const b = parseInt(message.match(/block=(\d+)/)[1], 10);
354 const from = parseInt(message.match(/from=(\d+)/)[1], 10);
355 const to = parseInt(message.match(/to=(\d+)/)[1], 10);
356 if (currentBlocks[b]) currentBlocks[b].push({ from, to });
357 return;
358 }
359 if (message.startsWith('BLOCKS_END')) {
360 if (currentBlocks) {
361 blocks = currentBlocks;
362 edgeToBlock = buildEdgeToBlock(blocks);
363 currentBlocks = null;
364 }
365 return;
366 }
367
368 // --- BCT ---
369 if (message.startsWith('BCT_BEGIN')) {
370 const m = message.match(/root=(-?\d+)/);
371 currentBCT = {
372 nodes: [],
373 edges: [],
374 root: m ? parseInt(m[1], 10) : null
375 };
376 return;
377 }
378 if (message.startsWith('BCT_NODE')) {
379 if (!currentBCT) return;
380 const id = parseInt(message.match(/id=(\d+)/)[1], 10);
381 const kind = message.match(/kind=(\w+)/)[1];
382 const value = parseInt(message.match(/value=(\d+)/)[1], 10);
383 currentBCT.nodes.push({ id, kind, value });
384 return;
385 }
386 if (message.startsWith('BCT_EDGE')) {
387 if (!currentBCT) return;
388 const from = parseInt(message.match(/from=(\d+)/)[1], 10);
389 const to = parseInt(message.match(/to=(\d+)/)[1], 10);
390 const type = parseInt(message.match(/type=(\d+)/)[1], 10);
391 currentBCT.edges.push({ from, to, type });
392 return;
393 }
394 if (message.startsWith('BCT_END')) {
395 if (currentBCT) {
396 bcts.push(currentBCT);
397 currentBCT = null;
398 }
399 return;
400 }
401
402 // --- FPQ ---
403 if (message.startsWith('FPQ_BEGIN')) {
404 const m = message.match(/root=(-?\d+)/);
405 currentFPQ = {
406 nodes: [],
407 edges: [],
408 root: m ? parseInt(m[1], 10) : null
409 };
410 return;
411 }
412 if (message.startsWith('FPQ_NODE')) {
413 if (!currentFPQ) return;
414 const id = parseInt(message.match(/id=(\d+)/)[1], 10);
415 const type = message.match(/type=(\w+)/)[1];
416 const blockMatch = message.match(/block=(\d+)/);
417 const valueMatch = message.match(/value=(\d+)/);
418 currentFPQ.nodes.push({
419 id,
420 type, // 'P' | 'F_BLOCK' | 'F_GADGET' | 'LEAF'
421 blockIndex: blockMatch ? parseInt(blockMatch[1], 10) : -1,
422 value: valueMatch ? parseInt(valueMatch[1], 10) : -1
423 });
424 return;
425 }
426 if (message.startsWith('FPQ_EDGE')) {
427 if (!currentFPQ) return;
428 const from = parseInt(message.match(/from=(\d+)/)[1], 10);
429 const to = parseInt(message.match(/to=(\d+)/)[1], 10);
430 const pos = parseInt(message.match(/pos=(\d+)/)[1], 10);
431 currentFPQ.edges.push({ from, to, pos });
432 return;
433 }
434 if (message.startsWith('FPQ_END')) {
435 if (currentFPQ) {
436 fpqs.push(currentFPQ);
437 currentFPQ = null;
438 }
439 return;
440 }
441
442 // --- Layout / counts ---
443 if (message.startsWith('RESULT: ')) {
444 let layout = message.replace('RESULT: ', '').split(' ').map(item => item.trim());
445 layout.pop(); // strip the trailing empty token.
446 layouts.push(layout);
447 return;
448 }
449 if (message.startsWith('NUMBER OF LAYOUTS: ')) {
450 numberOfLayouts = parseInt(message.replace('NUMBER OF LAYOUTS: ', '').split(' ').map(item => item.trim())[0], 10);
451 updateStatistics();
452 return;
453 }
454}

463function buildEdgeToBlock(blocksArr) {
464 const map = new Map();
465 for (let i = 0; i < blocksArr.length; i++) {
466 for (const e of blocksArr[i]) {
467 const a = Math.min(e.from, e.to);
468 const b = Math.max(e.from, e.to);
469 map.set(a + '-' + b, i);
470 }
471 }
472 return map;
473}

482function blockOfEdge(from, to) {
483 const a = Math.min(parseInt(from, 10), parseInt(to, 10));
484 const b = Math.max(parseInt(from, 10), parseInt(to, 10));
485 const v = edgeToBlock.get(a + '-' + b);
486 return (v === undefined) ? -1 : v;

522// =============================================================================
523//
524// Vertices are laid out on a horizontal line in the order given by `layout`,
525// edges are drawn as curves above (or below) the line by vis-network's

528// the color used for that block's node in the BCT and FPQ panels. If the

539function drawLayout(graph, layout) {
540 const nodes = new vis.DataSet();
541 const edges = new vis.DataSet();
542 const mapping = new Map();
543
544 let i = 0;
545 layout.forEach(label => { mapping.set(label, i); i += 1; });
546
547 // VERTICAL spine: nodes stacked top-to-bottom (x = 0); arcs bow to the RIGHT.
548 const V = 75;
549 graph.nodes.forEach(node => {
550 nodes.add({ id: mapping.get(node.label), label: node.label, x: 0, y: mapping.get(node.label) * V });
551 });
552
553 graph.edges.forEach(edge => {
554 const b = blockOfEdge(edge.from, edge.to);
555 const c = (b >= 0) ? blockColor(b) : 'black';
556 edges.add({
557 from: mapping.get(edge.from),
558 to: mapping.get(edge.to),
559 color: { color: c, highlight: c, hover: c },
560 width: 2
561 });
562 });
563
564 // Reserve room on the RIGHT so the arcs are not clipped; the spine then sits
565 // on the left of the panel. HEAD ~ the widest arc's rightward extent.
566 let maxSpan = 0;
567 graph.edges.forEach(edge => {
568 const a = mapping.get(edge.from), bb = mapping.get(edge.to);
569 if (a != null && bb != null) maxSpan = Math.max(maxSpan, Math.abs(a - bb));
570 });
571 const HEAD = maxSpan * V * 0.55; // tune if arcs clip / too much room
572 if (HEAD > 0) {
573 const midY = (i - 1) * V / 2;
574 const invisible = {
575 size: 0, shape: 'dot', label: '',
576 color: { background: 'rgba(0,0,0,0)', border: 'rgba(0,0,0,0)' },
577 physics: false, fixed: { x: true, y: true }
578 };
579 nodes.add(Object.assign({ id: '__padRight', x: HEAD, y: midY }, invisible));
580 nodes.add(Object.assign({ id: '__padLeft', x: -HEAD * 0.10, y: midY }, invisible));
581 }

610// =============================================================================
611// BCT drawing (bottom-left panel)
612// =============================================================================
613//
614// We use a manual "tidy tree" layout (same algorithm as drawFPQ) instead of
615// vis-network's hierarchical layout, because we want absolute control over
616// the left-to-right order of children: the C++ side emits BCT_EDGE entries
617// in the order dictated by the current permutation (see
618// orderedChildrenOfCutpoint in main_enum.cpp), and we must preserve that
619// order so flipping through layouts visibly rearranges the BCT in lockstep
620// with the layout itself.
621//
622// Graphical conventions:
623// - Block nodes (biconnected components):
624// rectangle filled with the block's chromatism color, label "B<i>".
625// Same color is used for the corresponding F_NODE_BLOCK in the FPQ
626// panel and for the edges of that block in the layout panel.
627// - Cutpoint nodes (cut vertices):
628// ellipse with the standard cyan background, label "<v>".
629// - Edge colors encode the role of the cutpoint in the child block,
630// deliberately chosen dark/saturated to contrast with the pastel
631// block palette:
632// type=0 (source cutpoint) -> dark green "s"
633// type=1 (intermediate cutpoint) -> dark red "i"
634// type=2 (sink cutpoint) -> dark blue "t"
635//
636// A small legend is rendered below the network area so the meaning of the
637// edge colors is self-documenting. The legend is inserted via inline CSS
638// (no need to touch styles.css).
639//
640// The `BCT_EDGE_STYLE` map itself is declared near the top of the file
641// (it is needed during the initial drawBCT call to render the legend).
642// =============================================================================

816// =============================================================================
817//
818// Graphical conventions (kept consistent with the BCT panel where it makes
819// sense, to ease cross-reading):
820// - F_BLOCK (one for each biconnected block) -> rectangle filled with the
821// block's chromatism color,
822// label "B<i>"
823// - F_GADGET (cutpoint gadget wrapper) -> small empty grey rectangle
824// - P_NODE (free permutation of children) -> light green ellipse, "P"
825// - LEAF, cutpoint vertex -> cyan ellipse, "<v>"
826// - LEAF, regular vertex -> bare text, "<v>"
827//
828// To distinguish "cutpoint leaf" from "regular leaf" we need to know which
829// graph vertices are cutpoints: we recover that from the corresponding BCT
830// passed as the second argument. By construction `bcts[i]` and `fpqs[i]`
831// are paired.
832//
833// LAYOUT: explicit (x, y) coordinates are computed via the classic recursive
834// "tidy tree" / subtree-width algorithm; vis-network's own hierarchical
835// layout is disabled. The FPQ-tree is a strict tree (one parent per node,
836// no cycles), so this layout produces zero crossings. It also respects the
837// `pos` ordering of children, which is significant for F-nodes (and for
838// P-nodes it encodes the current permutation).
839//
840// Algorithm:
841// - every leaf occupies one horizontal slot;
842// - every internal node is centered on the midpoint between its first and
843// its last child;
844// - children are visited in order of `pos`.
845// =============================================================================

1223// =============================================================================
1224// Ranking / Unranking controls
1225// =============================================================================
1226//
1227// The C++ enumerator (see main_enum.cpp / main()) emits every layout in a
1228// fixed canonical order:
1229//
1230// for each rooting rho in `rootings` (in order):
1231// for each permutation pi (lexicographic, via std::next_permutation
1232// on the cutpoint children in BFS order):
1233// emit layout(rho, pi)
1234//
1235// Therefore the position of a layout in the parallel arrays `layouts[]`,
1236// `bcts[]`, `fpqs[]` IS its rank. Consequences (all O(1), pure JS, no need to
1237// touch the C++):
1238// - RANK (current layout -> integer): `currentIndex`.
1239// - UNRANK (integer k -> layout): set `currentIndex = k` and redraw.
1240//
1241// The set of layouts sharing the same rooting is exactly one contiguous run in
1242// the arrays. Grouping `bcts[]` by the *root block* of each entry recovers the
1243// rootings: each run is one admissible root block, and its length is the number
1244// of permutations of that rooting (N_rho = product of the factorials of the
1245// P-node sizes). This is what feeds the segmented bar:
1246// - one segment per admissible root block,
1247// - segment width proportional to its layout count,
1248// - segment colored with the block-chromatism color (same color the block
1249// gets in the BCT and as F_BLOCK in the FPQ tree),
1250// - segment labeled "B<i>" with its inclusive rank range "<start>-<end>".
1251//
1252// Identifiers shown in the UI are 1-based, matching the "layout X of Y"
1253// navigation bar at the bottom: the rank readout, the unrank range and the
1254// segment ranges all use the same numbering as the arrows, so the numbers line
1255// up exactly. Internally `currentIndex` and the group start/end stay 0-based
1256// (they index the parallel arrays); the +1 is applied only at display time, and
1257// the unrank input is converted back with -1. (Note: the Di Battista/Grosso/
1258// Maragno/Patrignani ranking paper uses 0-based ranks; switching back is just a
1259// matter of dropping the +1/-1 offsets here.)
1260// =============================================================================

1317function rebuildRootingGroupsIfNeeded() {
1318 if (bcts.length === rootingGroupsForLength && rootingGroups.length > 0) return;
1319 rootingGroupsForLength = bcts.length;
1320 rootingGroups = [];
1321 for (let i = 0; i < bcts.length; i++) {
1322 const rb = rootBlockValueOf(bcts[i], i);
1323 const last = rootingGroups[rootingGroups.length - 1];
1324 if (!last || last.rootBlock !== rb) {
1325 rootingGroups.push({ rootBlock: rb, start: i, count: 1 });
1326 } else {
1327 last.count++;
1328 }
1329 }
1330 for (const g of rootingGroups) g.end = g.start + g.count - 1;
1331}

1339function decomposeRank(k) {
1340 for (let r = 0; r < rootingGroups.length; r++) {
1341 const g = rootingGroups[r];
1342 if (k >= g.start && k <= g.end) {
1343 return { rooting: r, perm: k - g.start, rootBlock: g.rootBlock, count: g.count };
1344 }
1345 }
1346 return null;
1347}

1355function composeRank(rooting, perm) {
1356 if (rooting < 0 || rooting >= rootingGroups.length) return -1;
1357 const g = rootingGroups[rooting];
1358 if (perm < 0 || perm >= g.count) return -1;
1359 return g.start + perm;
