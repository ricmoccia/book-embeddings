264 lastUsedPermutations = permutations;

267 std::vector<int> order;
268 std::queue<TreeNode*> queue;
269 queue.push(root);
270
271 while (!queue.empty()) {
272 TreeNode* u = queue.front();
273 queue.pop();
274
275 if (u->isCutpoint) {
276 for (int childType: {0, 2}) {
277 std::vector<TreeNode*> children = u->getChildren(treeId, childType);
278
279 if (children.size() > 1) {
280 // mergeLayouts() splices each SOURCE (type-0) child block immediately
281 // after the cutpoint, which reverses the processing order in the
282 // spine. Enqueue source children in REVERSE permutation order so the
283 // resulting spine reads left-to-right in lexicographic order, matching
284 // the BCT and FPQ panels. Sink (type-2) children keep their order.
285 if (childType == 0) {
286 for (auto rit = permutations[u->value][childType].rbegin();
287 rit != permutations[u->value][childType].rend(); ++rit) {
288 queue.push(children[*rit]);
289 }
290 } else {
291 for (int indexOfChild: permutations[u->value][childType]) {
292 queue.push(children[indexOfChild]);
293 }
294 }
295
296 // Advance the odometer: move this cutpoint's permutation to the
297 // next one; on wrap-around reset it and carry to the next cutpoint.
298 // (This is what enumerates ALL layouts of a rooting, not just the first.)
299 if (toPermute) {
300 if (std::next_permutation(permutations[u->value][childType].begin(), permutations[u->value][childType].end())) {
301 toPermute = false;
302 }
303 else {
304 std::vector<int> firstPermutation;
305 for (int i = 0; i < (int)permutations[u->value][childType].size(); i++) {
306 firstPermutation.push_back(i);
307 }
308 permutations[u->value][childType] = firstPermutation;
309 }
310 }
311 } else {
312 for (TreeNode* child: children) {
313 queue.push(child);
314 }
315 }
316 }
317 }
318 else {
319 // A block node: emit it and enqueue all children regardless of type.
320 order.push_back(u->value);
321 for (int childType = 0; childType < 3; childType++) {
322 for (TreeNode* child: u->getChildren(treeId, childType)) {
323 queue.push(child);
324 }
325 }
326 }
327 }
328
329 if (toPermute) {
330 // No cutpoint had a permutation to advance: this was the last one.
331 hasNext = false;
332 }
333 return order;
334 }

952 std::unordered_map<int, bool> visited;
953 std::queue<TreeNode*> queue;
954
955 // Order of the components.
956 std::vector<int> orderOfComponents;
957
958 visited[rootOfBCT->id] = true;
959 queue.push(rootOfBCT);
960
961 bool existsRooting = rootOfBCT->hasRooting(rootOfBCT->value);
962
963 // BFS over the un-rooted BCT to root it and find the component order
964 // (degree of freedom).
965 while (!queue.empty()) {
966 TreeNode* u = queue.front();
967 queue.pop();
968 if (!u->isCutpoint) {
969 orderOfComponents.push_back(u->value);
970 }
971
972 // Visit every neighbor and record children.
973 for (TreeNode* neighbor : u->neighbors) {
974 if (!visited[neighbor->id]) {
975 visited[neighbor->id] = true;
976 queue.push(neighbor);
977 if (!existsRooting) {
978 int childrenType = -1;
979 if (neighbor->isCutpoint) {
980 if (sourceAndSinkOfComponents[u->value][0]->index() == neighbor->value) {
981 childrenType = 0; // source
982 }
983 else if (sourceAndSinkOfComponents[u->value][1]->index() == neighbor->value) {
984 childrenType = 2; // sink
985 }
986 else {
987 childrenType = 1; // intermediate
988 }
989 }
990 else {
991 if (sourceAndSinkOfComponents[neighbor->value][0]->index() == u->value) {
992 childrenType = 0;
993 }
994 else if (sourceAndSinkOfComponents[neighbor->value][1]->index() == u->value) {
995 childrenType = 2;
996 }
997 else {
998 childrenType = 1;
999 }
1000 neighbor->setParent(rootOfBCT->id, u);
1001 }
1002 u->addChild(rootOfBCT->id, neighbor, childrenType);
1003 }
1004 }
1005 }
1006 }
1007 return orderOfComponents;

1191 // FPQ-tree construction from a rooted BCT
1192 //
1193 // Reference: BCT2FPQtree.pptx (R.Moccia).
1194 //
1195 // Two mutually recursive helpers:
1196 //
1197 // buildFPQBlock(B, parentCutpointVertex):
1198 // iterates over the local topological order of B (=
1199 // topologicalOrders[B->value]), skipping the parent cutpoint
1200 // (already represented in the parent gadget). Each cutpoint that
1201 // has BCT-children becomes a gadget; the other vertices become
1202 // leaves. Everything is wrapped in an F_NODE_BLOCK.
1203 //
1204 // buildFPQCutpointGadget(c):
1205 // separates the BCT children of c into
1206 // - type-2 (cutpoint is the SINK of the block -> block on the LEFT of c)
1207 // - type-0 (cutpoint is the SOURCE of the block -> block on the RIGHT of c)
1208 // expands each recursively with buildFPQBlock(., c). Builds
1209 // F_NODE_GADGET( [P(type2)?], LEAF(c), [P(type0)?] )
1210 // with the three simplifications from slide 11:
1211 // - empty side -> the P is omitted;
1212 // - single child on a side -> no P, that subtree directly;
1213 // - non-cutpoint vertices in B -> plain leaves (handled in buildFPQBlock).
1214 //
1215 // `currentRootId` (= root->id of the currently rooted BCT) is used to
1216 // query TreeNode::getChildren / getParent. `nextFPQId` and
1217 // `allFPQNodes` keep track of created ids/nodes for cleanup.
1218 //
1219 // `currentPermutations` is the permutation state of the
1220 // BCTPermutationEnumerator: it determines the left-to-right order of
1221 // the children of a P-node (so the FPQ visualization reflects the
1222 // permutation that produced the current layout).
1223 // =========================================================================

1262 FPQNode* buildFPQBlock(TreeNode* blockNode, int parentCutpointVertex, int currentRootId,
1263 const std::unordered_map<int, std::unordered_map<int, std::vector<int>>>& currentPermutations) {
1264 int blockIdx = blockNode->value;
1265 Array<int, int> localOrder = topologicalOrders[blockIdx];
1266
1267 // Map: vertex_index -> TreeNode* of the cutpoint, but only for the
1268 // cutpoints that are CHILDREN of blockNode in this rooting (and thus
1269 // must be expanded into gadgets). For a parent block, cutpoint children
1270 // can occur with any childType (0/1/2) according to the role of the
1271 // cutpoint in the block.
1272 std::unordered_map<int, TreeNode*> cutpointChildOf;
1273 for (int childType = 0; childType < 3; childType++) {
1274 for (TreeNode* child : blockNode->getChildren(currentRootId, childType)) {
1275 if (child->isCutpoint) {
1276 cutpointChildOf[child->value] = child;
1277 }
1278 }
1279 }
1280
1281 std::vector<FPQNode*> fChildren;
1282 for (int i = 0; i < localOrder.size(); i++) {
1283 int v = localOrder[i];
1284
1285 // (1) Skip the parent cutpoint: it is already represented in the gadget above.
1286 if (v == parentCutpointVertex) continue;
1287
1288 // (2) If v is a cutpoint child of blockNode in this rooting, expand it as a gadget.
1289 auto it = cutpointChildOf.find(v);
1290 if (it != cutpointChildOf.end()) {
1291 FPQNode* gadget = buildFPQCutpointGadget(it->second, currentRootId, currentPermutations);
1292 fChildren.push_back(gadget);
1293 } else {
1294 // (3) Plain vertex (not a cutpoint, or cutpoint with no children
1295 // in this rooting -- impossible by BCT definition for a cutpoint child).
1296 fChildren.push_back(makeFPQLeaf(v));
1297 }
1298 }
1299
1300 // Note: a previous version returned fChildren[0] directly when size()==1,
1301 // to "elide" degenerate blocks. We now always keep the F_NODE_BLOCK so we
1302 // preserve a 1-to-1 correspondence between blocks and F-nodes, which makes
1303 // the resulting tree easier to read.
1304
1305 FPQNode* f = makeFPQNode(FPQType::F_NODE_BLOCK);
1306 f->blockIndex = blockIdx;
1307 f->children = fChildren;
1308 return f;
1309 }

1320 FPQNode* buildFPQCutpointGadget(TreeNode* cutpointNode, int currentRootId,
1321 const std::unordered_map<int, std::unordered_map<int, std::vector<int>>>& currentPermutations) {
1322 int cVertex = cutpointNode->value;
1323
1324 // type-2: cutpoint is the SINK of the child block -> children on the LEFT of c.
1325 std::vector<TreeNode*> type2Blocks = cutpointNode->getChildren(currentRootId, 2);
1326 // type-0: cutpoint is the SOURCE of the child block -> children on the RIGHT of c.
1327 std::vector<TreeNode*> type0Blocks = cutpointNode->getChildren(currentRootId, 0);
1328
1329 // Apply the current permutation, if any, to reorder these vectors.
1330 // The permutation entries from BCTPermutationEnumerator are keyed
1331 // by cutpoint VALUE (graph vertex index) and edge type (0 or 2).
1332 auto applyPermutation = [&](std::vector<TreeNode*>& vec, int type) {
1333 if (vec.size() <= 1) return;
1334 auto itC = currentPermutations.find(cVertex);
1335 if (itC == currentPermutations.end()) return;
1336 auto itT = itC->second.find(type);
1337 if (itT == itC->second.end()) return;
1338 const std::vector<int>& perm = itT->second;
1339 if (perm.size() != vec.size()) return;
1340 std::vector<TreeNode*> permuted;
1341 permuted.reserve(vec.size());
1342 for (int idx : perm) permuted.push_back(vec[idx]);
1343 vec = permuted;
1344 };
1345 applyPermutation(type2Blocks, 2);
1346 applyPermutation(type0Blocks, 0);
1347
1348 FPQNode* gadget = makeFPQNode(FPQType::F_NODE_GADGET);
1349
1350 // --- Left side (type-2) ---
1351 if (!type2Blocks.empty()) {
1352 std::vector<FPQNode*> leftSubtrees;
1353 for (TreeNode* childBlock : type2Blocks) {
1354 leftSubtrees.push_back(buildFPQBlock(childBlock, cVertex, currentRootId, currentPermutations));
1355 }
1356 if (leftSubtrees.size() == 1) {
1357 gadget->addChild(leftSubtrees[0]); // simplification: no P when single child
1358 } else {
1359 FPQNode* p = makeFPQNode(FPQType::P_NODE);
1360 p->children = leftSubtrees;
1361 gadget->addChild(p);
1362 }
1363 }
1364
1365 // --- The cutpoint c itself ---
1366 gadget->addChild(makeFPQLeaf(cVertex));
1367
1368 // --- Right side (type-0) ---
1369 if (!type0Blocks.empty()) {
1370 std::vector<FPQNode*> rightSubtrees;
1371 for (TreeNode* childBlock : type0Blocks) {
1372 rightSubtrees.push_back(buildFPQBlock(childBlock, cVertex, currentRootId, currentPermutations));
1373 }
1374 if (rightSubtrees.size() == 1) {
1375 gadget->addChild(rightSubtrees[0]);
1376 } else {
1377 FPQNode* p = makeFPQNode(FPQType::P_NODE);
1378 p->children = rightSubtrees;
1379 gadget->addChild(p);
1380 }
1381 }
1382
1383 return gadget;
1384 }

1522 std::vector<int> findAdmissibleRoots() {
1523 std::vector<int> admissible;
1524
1525 for (int blockIdx = 0; blockIdx < numberOfBiconnectedComponents; blockIdx++) {
1526
1527 // ----------- Condition (1): s(B) must be a global source -----------
1528 // s(B) is a global source iff in every block containing it, it
1529 // is the local source (i.e. all BCT edges incident to s(B) are
1530 // labelled 0). We reuse sourceAndSinkOfComponents and
1531 // componentsOfNode which are already populated.
1532 int srcIndex = sourceAndSinkOfComponents[blockIdx][0]->index();
1533 bool isGlobalSource = true;
1534 for (int compIdx : componentsOfNode[srcIndex]) {
1535 if (sourceAndSinkOfComponents[compIdx][0]->index() != srcIndex) {
1536 // In some other block s(B) is not the source
1537 // (it is intermediate or sink) -> some block "always comes before".
1538 isGlobalSource = false;
1539 break;
1540 }
1541 }
1542 if (!isGlobalSource) {
1543 std::cout << "block " << blockIdx << " discarded: its source "
1544 << srcIndex << " is not a global source" << std::endl;
1545 continue;
1546 }
1547
1548 // ----------- Condition (2): no restricted cutpoint when rooting at B -----------
1549 // Rooting the BCT at blockIdx, verify that no cutpoint has a
1550 // block child attached with label 1 (no block "always lies above"
1551 // via paths 2-0-2-0... ending with a 1-edge).
1552 TreeNode* rootTreeNode = treeNodeOfComponent[blockIdx];
1553 rootBCT(rootTreeNode); // idempotent
1554
1555 bool noRestriction = true;
1556 std::queue<TreeNode*> bfsQueue;
1557 bfsQueue.push(rootTreeNode);
1558 while (!bfsQueue.empty()) {
1559 TreeNode* u = bfsQueue.front();
1560 bfsQueue.pop();
1561 if (u->isCutpoint) {
1562 // If u (cutpoint) has a block child attached with label 1
1563 // then u is a restricted cutpoint: B is not admissible.
1564 if (!u->getChildren(blockIdx, 1).empty()) {
1565 noRestriction = false;
1566 break;
1567 }
1568 }
1569 for (int childType = 0; childType < 3; childType++) {
1570 for (TreeNode* child : u->getChildren(blockIdx, childType)) {
1571 bfsQueue.push(child);
1572 }
1573 }
1574 }
1575
1576 if (noRestriction) {
1577 admissible.push_back(blockIdx);
1578 std::cout << "block " << blockIdx << " is an admissible root" << std::endl;
1579 }
1580 else {
1581 std::cout << "block " << blockIdx << " discarded: rooting it would leave "
1582 << "a restricted cutpoint in the BCT" << std::endl;
1583 }
1584 }
1585
1586 return admissible;
1587 }
